#Creamos las tablas


CREATE TABLE articulo (
	id_noticia int not null auto_increment,
	fecha varchar(20),
	titulo varchar(200),
	texto_noticia varchar(5000),
	url varchar(200),
	sentiment_value int,
	primary key (id_noticia)
);

CREATE TABLE tags (
	id_tag int not null auto_increment,
	id_noticia int,
	visitas varchar(2000),
	name_tag varchar(60),
	primary key (id_tag)
);
